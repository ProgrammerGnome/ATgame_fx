import java.util.Arrays;

public class Main {
    public static void main(String[] args) {
        int[][] array1 = {{0, 1}, {4, 1}, {2, 5}};
        int[][] array2 = {{4, 1}, {0, 1}, {2, 5}};

        boolean areEqual = Arrays.deepEquals(array1, array2);
        System.out.println("Are arrays equal? " + areEqual);
    }
}
